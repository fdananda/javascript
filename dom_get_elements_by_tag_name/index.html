DOM getElementsByTagName()

O método getElementsByTagName() retorna uma coleção de todos os elementos com um nome de tag especificado. O getElementsByTagName("*")retorna todos os elementos do documento.

Exemplo:

  • li 1
  • li 2
  • li 3
  • li 4
  • li 5

function obterElementosPorNomeDeTag() {

	let colecaoTagName = document.getElementsByTagName('li')
	let saida = document.getElementById("exemplos")

	let textoSaida = `A colecaoTagName é uma: ${colecaoTagName}

	O tamanho da colecaoTagName é: ${colecaoTagName.length}

	`

	for (let i =0; i < colecaoTagName.length; i++) {	
	textoSaida += `O item ${i} é ${colecaoTagName[i].textContent}
`
	saida.innerHTML = textoSaida
	}
}
