Estruturas condicionais/Conditional Statements - if... else / switch / operador ternário

A instrução if especifica um bloco de código a ser executado, se uma condição especificada for verdadeira. A instrução else especifica um bloco de código a ser executado, se a mesma condição for falsa. O else if especifica uma nova condição para testar se a condição anterior for falsa.

A instrução switch executa um bloco de código dependendo de diferentes casos. Essa é a solução perfeita para instruções if/else longas e aninhadas.

O Operador Ternário funciona como uma versão simplificada da instrução if else, com a estrutura formada por: condicao ? valor verdadeiro : valor falso

Exemplo: 

function chamarIfElse() {

    let idade = 45
    let textoSaida = ""

    if (idade <= 18) {
        textoSaida = "Você é criança"
    }else if(idade > 18 && idade < 60){
        textoSaida = "Você é adulto"
    }else if (idade >= 60 && idade < 120){
        textoSaida = "Você é idoso"
    }else{
        textoSaida = "Você é uma lenda"
    }

	let saida = document.getElementById("exemplos")
	saida.innerHTML = textoSaida
}

function chamarSwitch() {

    let textoSaida = ""

    switch (new Date().getDay()) {
      case 0:
        textoSaida = "Domingo";
        break;
      case 1:
        textoSaida = "Segunda-feira";
        break;
      case 2:
        textoSaida = "Terça-feira";
        break;
      case 3:
        textoSaida = "Quarta-feira";
        break;
      case 4:
        textoSaida = "Quinta-feira";
        break;
      case 5:
        textoSaida = "Sexta-feira";
        break;
      case 6:
        textoSaida = "Sábado";
        break;
      default:
        textoSaida = "ops...";
    }

    let saida = document.getElementById("exemplos")
    saida.innerHTML = "Hoje é " + textoSaida
}

function chamarOperadorTernario() {

    let nome = "Fernand"
    let textoSaida = ""

    nome == "Fernanda" ? textoSaida = "O nome escolhido é Fernanda" : textoSaida = "O nome escolhido não é Fernanda"

    let saida = document.getElementById("exemplos")
    saida.innerHTML = textoSaida
}
