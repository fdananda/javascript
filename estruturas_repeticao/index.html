Estruturas de repetição/loops - for / while / do... while

A instrução for define um bloco de código que é executado desde que uma condição seja true.

A instrução while cria um loop (em um bloco de código) que é executado enquanto uma condição é true.

A instrução do...while define um bloco de código a ser executado uma vez e repetido enquanto uma condição for true. É usado quando se deseja executar um bloco de código pelo menos uma vez.

Exemplo: 

function chamarFor() {
    let nomes = ["Fernanda", "Helena", "Rodrigues", "Moreira"]
    let listaNome = ""
    
    for (i=0; i < nomes.length; i++) {
        listaNome += "+ " + nomes[i] + "
"
    }

    let saida = document.getElementById("exemplos")
        saida.innerHTML = "Resultado da instrução for: 
" + listaNome
}

function chamarWhile() {
    let nomes = ["Fernanda", "Helena", "Rodrigues", "Moreira"]
    let listaNome = ""
    let i = 0;

    while (i < nomes.length) {
     listaNome += "+ " + nomes[i] + "
"
     i++;
    }

    let saida = document.getElementById("exemplos")
        saida.innerHTML = "Resultado da instrução while: 
" + listaNome
}


function chamarDoWhile() {
    let nomes = ["Fernanda", "Helena", "Rodrigues", "Moreira"]
    let listaNome = ""
    let i = 0;

    do {
     listaNome += "+ " + nomes[i] + "
"
     i++;
    }
    while (i < nomes.length);

    let saida = document.getElementById("exemplos")
        saida.innerHTML = "Resultado da instrução do... while: 
" + listaNome
}
