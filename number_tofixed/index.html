Number toFixed()

O método toFixed() converte um número em uma string e arredonda a string para um número especificado de casas decimais.

Exemplo: 

	function converterParaString() {
	let numero1  	= 20/30
	let numero2 	= 3.14159
	
	let textoSaida = 
	"20/30 sem o toFixed: " + numero1 + 
	" 3.14159 sem o toFixed: " + numero2 + 
	" 20/30 com o toFixed(): " + numero1.toFixed() + 
	" 3.14159 com o toFixed(1) para uma casa decimal: " + numero2.toFixed(1) 

	let saida = document.getElementById("exemplos")
	saida.innerHTML = textoSaida
}
