try...catch...finally - throw

A combinação try...catch...finally trata os erros sem interromper a execução. A instrução try contém o bloco de código a ser executado que pode (ou não) gerar um erro. A instrução catch define um bloco de código que será executado caso um erro seja capturado. A instrução finally define um bloco de código a ser executado independentemente do resultado. A instrução throw define um erro personalizado.

Exemplo: 

let saida = document.getElementById("exemplos")

function gerarTryCatch() {   

    let saida2 = document.getElementById("exemplos2")

    try{
      //alert("Hello World!") - CÓDIGO CORRETO
      alerta("Hello World!") //CÓDIGO ERRADO
    }catch(erro){ 
      tratarErro(erro)
    }finally{
      saida2.innerHTML = `Finalizado`
    }
}

function tratarErro( erro ){
  if (erro.message == "alerta is not defined") {
    throw new Error(saida.innerHTML = "Função 'alerta' não existe") 
  }else{
    throw new Error(saida.innerHTML = "Um erro aconteceu") 
  }
}
